6. Authority Rule — Gap vs. Re-Derivation 
20 mismatches were found between gold standard and gap answers in SYN_0001. Some are legitimate (gold is 
more clinically accurate), but most represent the LLM ignoring the gap answers and re-deriving from documents. 
The new authority rule eliminates re-derivation for any field present in gap_context.

Recommended Post-Processing Validator 
Even with corrected prompts, a Python validator on the gold standard output catches any remaining arithmetic slip. 

 
def validate_gold_standard(gold: dict) -> list[str]: 
    errors = [] 
    # Forbidden keys 
    forbidden = {'PHQ_MOOD_INTERVIEW', 'ALLERGIES', 'VITAL_SIGNS', 'CIRCULATORY_HISTORY'} 
    for k in gold: 
        if k in forbidden: 
            errors.append(f'Forbidden key in gold: {k}') 
    # BIMS arithmetic 
    if all(k in gold for k in ['C0300A','C0300B','C0300C','C0300']): 
        exp = int(gold['C0300A'])+int(gold['C0300B'])+int(gold['C0300C']) 
        if int(gold['C0300']) != exp: errors.append(f'C0300 mismatch: {gold["C0300"]} != 
{exp}') 
    if all(k in gold for k in ['C0400A','C0400B','C0400C','C0400']): 
        exp = int(gold['C0400A'])+int(gold['C0400B'])+int(gold['C0400C']) 
        if int(gold['C0400']) != exp: errors.append(f'C0400 mismatch: {gold["C0400"]} != 
{exp}') 



    if all(k in gold for k in ['C0200','C0300','C0400','C0500']): 
        exp = int(gold['C0200'])+int(gold['C0300'])+int(gold['C0400']) 
        if int(gold['C0500']) != exp: errors.append(f'C0500 mismatch: {gold["C0500"]} != 
{exp}') 
    # PHQ gate 
    a1 = int(gold.get('D0150A1', 0)); b1 = int(gold.get('D0150B1', 0)) 
    if a1+b1 < 3: 
        a2 = int(gold.get('D0150A2',0)) if a1==1 else 0 
        b2 = int(gold.get('D0150B2',0)) if b1==1 else 0 
        if str(gold.get('D0160')) != str(a2+b2): 
            errors.append(f'D0160 gate violation: got {gold["D0160"]}, should be {a2+b2}') 
    # N0415 completeness 
    for sub in 'ABCDEFGHI': 
        if f'N0415{sub}' not in gold: 
            errors.append(f'N0415{sub} missing from gold standard') 
    # GG admission completeness 
    must_have = 
['GG0130D1','GG0130E1','GG0170A1','GG0170B1','GG0170C1','GG0170D1','GG0170E1','GG0170F1'] 
    for k in must_have: 
        if k not in gold: errors.append(f'{k} missing from gold standard') 
    return errors
